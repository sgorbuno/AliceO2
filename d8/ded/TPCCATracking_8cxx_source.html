67 int TPCCATracking::convertClusters(TChain* inputClustersChain, const std::vector<o2::TPC::Cluster>* inputClustersArray,
68  ClusterNativeAccessFullTPC& outputClusters,
69  std::unique_ptr<ClusterNative[]>& clusterMemory)
70 {
71  if (inputClustersChain && inputClustersArray) {
72  LOG(FATAL) << "Internal error, must not pass in both TChain and TClonesArray of clusters\n";
73  }
74  int numChunks;
75  if (inputClustersChain) {
76  inputClustersChain->SetBranchAddress("TPCClusterHW", &inputClustersArray);
77  numChunks = inputClustersChain->GetEntries();
78  } else {
79  numChunks = 1;
80  }
81 
82  // index to the pointers in the store, used to assign converted clusters and sort them
83  // while outputClusters object only holds the const pointer for read access
84  ClusterNative* clusterptrs[Sector::MAXSECTOR][Constants::MAXGLOBALPADROW];
85  memset(clusterptrs, 0, sizeof(ClusterNative*) * Sector::MAXSECTOR * Constants::MAXGLOBALPADROW);
86  Mapper& mapper = Mapper::instance();
87  for (int iter = 0; iter < 2; iter++) {
88  for (int i = 0; i < Sector::MAXSECTOR; i++) {
89  for (int j = 0; j < Constants::MAXGLOBALPADROW; j++) {
90  outputClusters.nClusters[i][j] = 0;
91  }
92  }
93 
94  unsigned int nClusters = 0;
95  for (int iChunk = 0; iChunk < numChunks; iChunk++) {
96  if (inputClustersChain) {
97  inputClustersChain->GetEntry(iChunk);
98  }
99  for (const auto& cluster : *inputClustersArray) {
100  const CRU cru(cluster.getCRU());
101  const Sector sector = cru.sector();
102  const PadRegionInfo& region = mapper.getPadRegionInfo(cru.region());
103  const int rowInSector = cluster.getRow() + region.getGlobalRowOffset();
104 
105  // in first iteration (== 0) only the clusters are counted
106  // clusters are converted in the second iteraton once store has been allocated
107  if (iter == 1) {
108  ClusterNative& oCluster = clusterptrs[sector][rowInSector][outputClusters.nClusters[sector][rowInSector]];
109  oCluster.setTimeFlags(cluster.getTimeMean(), 0);
110  oCluster.setPad(cluster.getPadMean());
111  oCluster.setSigmaTime(cluster.getTimeSigma());
112  oCluster.setSigmaPad(cluster.getPadSigma());
113  oCluster.qTot = cluster.getQ();
114  oCluster.qMax = cluster.getQmax();
115  }
116  outputClusters.nClusters[sector][rowInSector]++;
117  }
118  nClusters += inputClustersArray->size();
119  }
120  if (iter == 0) {
121  clusterMemory.reset(new ClusterNative[nClusters]);
122  unsigned int pos = 0;
123  for (int i = 0; i < Sector::MAXSECTOR; i++) {
124  for (int j = 0; j < Constants::MAXGLOBALPADROW; j++) {
125  clusterptrs[i][j] = &clusterMemory[pos];
126  outputClusters.clusters[i][j] = &clusterMemory[pos];
127  pos += outputClusters.nClusters[i][j];
128  }
129  }
130  }
131  }
132  for (int i = 0; i < Sector::MAXSECTOR; i++) {
133  for (int j = 0; j < Constants::MAXGLOBALPADROW; j++) {
134  std::sort(clusterptrs[i][j], clusterptrs[i][j] + outputClusters.nClusters[i][j],
135  ClusterNativeContainer::sortComparison);
136  }
137  }
138 
139  return (0);
140 }

142 int TPCCATracking::runTracking(const ClusterNativeAccessFullTPC& clusters, std::vector<TrackTPC>* outputTracks,
143  MCLabelContainer* outputTracksMCTruth)
144 {
145  const static ParameterDetector& detParam = ParameterDetector::defaultInstance();
146  const static ParameterGas& gasParam = ParameterGas::defaultInstance();
147  const static ParameterElectronics& elParam = ParameterElectronics::defaultInstance();
148 
149  int nClusters[Sector::MAXSECTOR] = {};
150  int nClustersConverted = 0;
151  int nClustersTotal = 0;
152  float vzbin = (elParam.getZBinWidth() * gasParam.getVdrift());
153  float vzbinInv = 1.f / vzbin;
154 
155  Mapper& mapper = Mapper::instance();
156  for (int i = 0; i < Sector::MAXSECTOR; i++) {
157  for (int j = 0; j < Constants::MAXGLOBALPADROW; j++) {
158  if (clusters.nClusters[i][j] > 0xFFFF) {
159  LOG(ERROR) << "Number of clusters in sector " << i << " row " << j
160  << " exceeds 0xFFFF, which is currently a hard limit of the wrapper for the tracking!\n";
161  return (1);
162  }
163  nClusters[i] += clusters.nClusters[i][j];
164  }
165  nClustersTotal += nClusters[i];
166 
167  mClusterData[i].StartReading(i, nClusters[i]);
168 
169  for (int j = 0; j < Constants::MAXGLOBALPADROW; j++) {
170  Sector sector = i;
171  int regionNumber = 0;
172  while (j > mapper.getGlobalRowOffsetRegion(regionNumber) + mapper.getNumberOfRowsRegion(regionNumber))
173  regionNumber++;
174  CRU cru(sector, regionNumber);
175  const PadRegionInfo& region = mapper.getPadRegionInfo(cru.region());
176  for (int k = 0; k < clusters.nClusters[i][j]; k++) {
177  const ClusterNative& cluster = clusters.clusters[i][j][k];
178  AliHLTTPCCAClusterData& cd = mClusterData[i];
179  AliHLTTPCCAClusterData::Data& hltCluster = cd.Clusters()[cd.NumberOfClusters()];
180 
181  const float padY = cluster.getPad();
182  const int padNumber = int(padY);
183  const GlobalPadNumber pad = mapper.globalPadNumber(PadPos(j, padNumber));
184  const PadCentre& padCentre = mapper.padCentre(pad);
185  const float localY = padCentre.Y() - (padY - padNumber) * region.getPadWidth();
186  const float localYfactor = (cru.side() == Side::A) ? -1.f : 1.f;
187  float zPositionAbs = cluster.getTime() * vzbin;
188  if (!mTrackingCAO2Interface->GetParamContinuous())
189  zPositionAbs = detParam.getTPClength() - zPositionAbs;
190  else
191  zPositionAbs = sContinuousTFReferenceLength * vzbin - zPositionAbs;
192 
193  // sanity checks
194  if (zPositionAbs < 0 ||
195  (!mTrackingCAO2Interface->GetParamContinuous() && zPositionAbs > detParam.getTPClength())) {
196  LOG(INFO) << "Removing cluster " << i << " time: " << cluster.getTime() << ", abs. z: " << zPositionAbs
197  << "\n";
198  continue;
199  }
200 
201  hltCluster.fX = padCentre.X();
202  hltCluster.fY = localY * (localYfactor);
203  hltCluster.fZ = zPositionAbs * (-localYfactor);
204  hltCluster.fRow = j;
205  hltCluster.fAmp = cluster.qMax;
206  hltCluster.fFlags = cluster.getFlags();
207  hltCluster.fId = (i << 24) | (j << 16) | (k);
208 
209  cd.SetNumberOfClusters(cd.NumberOfClusters() + 1);
210  nClustersConverted++;
211  }
212  }
213  }
214 
215  if (nClustersTotal != nClustersConverted) {
216  LOG(INFO) << "Passed " << nClustersConverted << " (out of " << nClustersTotal << ") clusters to CA tracker\n";
217  }
218 
219  const AliHLTTPCGMMergedTrack* tracks;
220  int nTracks;
221  const AliHLTTPCGMMergedTrackHit* trackClusters;
222  int retVal = mTrackingCAO2Interface->RunTracking(mClusterData, tracks, nTracks, trackClusters);
223  if (retVal == 0) {
224  std::vector<std::pair<int, float>> trackSort(nTracks);
225  int tmp = 0, tmp2 = 0;
226  for (char cside = 0; cside < 2; cside++) {
227  for (int i = 0; i < nTracks; i++) {
228  if (tracks[i].OK() && tracks[i].CSide() == cside)
229  trackSort[tmp++] = { i, (cside == 0 ? 1.f : -1.f) * tracks[i].GetParam().GetZOffset() };
230  }
231  std::sort(trackSort.data() + tmp2, trackSort.data() + tmp,
232  [](const auto& a, const auto& b) { return (a.second > b.second); });
233  tmp2 = tmp;
234  if (cside == 0)
235  mNTracksASide = tmp;
236  }
237  nTracks = tmp;
238 
239  outputTracks->resize(nTracks);
240 
241  for (int iTmp = 0; iTmp < nTracks; iTmp++) {
242  auto& oTrack = (*outputTracks)[iTmp];
243  const int i = trackSort[iTmp].first;
244  float time0 = 0.f, tFwd = 0.f, tBwd = 0.f;
245 
246  float zHigh = 0, zLow = 0;
247  if (mTrackingCAO2Interface->GetParamContinuous()) {
248  float zoffset = tracks[i].CSide() ? -tracks[i].GetParam().GetZOffset() : tracks[i].GetParam().GetZOffset();
249  time0 = sContinuousTFReferenceLength - zoffset * vzbinInv;
250 
251  // estimate max/min time increments which still keep track in the physical limits of the TPC
252  zHigh = trackClusters[tracks[i].FirstClusterRef()].fZ - tracks[i].GetParam().GetZOffset(); // high R cluster
253  zLow = trackClusters[tracks[i].FirstClusterRef() + tracks[i].NClusters() - 1].fZ -
254  tracks[i].GetParam().GetZOffset(); // low R cluster
255 
256  bool sideHighA = (trackClusters[tracks[i].FirstClusterRef()].fNum >> 24) < Sector::MAXSECTOR / 2;
257  bool sideLowA =
258  (trackClusters[tracks[i].FirstClusterRef() + tracks[i].NClusters() - 1].fNum >> 24) < Sector::MAXSECTOR / 2;
259 
260  // calculate time bracket
261  float zLowAbs = zLow < 0.f ? -zLow : zLow;
262  float zHighAbs = zHigh < 0.f ? -zHigh : zHigh;
263  //
264  // tFwd = (Lmax - max(|zLow|,|zAbs|))/vzbin = drift time from cluster current Z till endcap
265  // tBwd = min(|zLow|,|zAbs|))/vzbin = drift time from CE till cluster current Z
266  //
267  if (zLowAbs < zHighAbs) {
268  tFwd = (detParam.getTPClength() - zHighAbs) * vzbinInv;
269  tBwd = zLowAbs * vzbinInv;
270  } else {
271  tFwd = (detParam.getTPClength() - zLowAbs) * vzbinInv;
272  tBwd = zHighAbs * vzbinInv;
273  }
274  }
275 
276  oTrack =
277  TrackTPC(tracks[i].GetParam().GetX(), tracks[i].GetAlpha(),
278  { tracks[i].GetParam().GetY(), tracks[i].GetParam().GetZ(), tracks[i].GetParam().GetSinPhi(),
279  tracks[i].GetParam().GetDzDs(), tracks[i].GetParam().GetQPt() },
280  { tracks[i].GetParam().GetCov(0), tracks[i].GetParam().GetCov(1), tracks[i].GetParam().GetCov(2),
281  tracks[i].GetParam().GetCov(3), tracks[i].GetParam().GetCov(4), tracks[i].GetParam().GetCov(5),
282  tracks[i].GetParam().GetCov(6), tracks[i].GetParam().GetCov(7), tracks[i].GetParam().GetCov(8),
283  tracks[i].GetParam().GetCov(9), tracks[i].GetParam().GetCov(10), tracks[i].GetParam().GetCov(11),
284  tracks[i].GetParam().GetCov(12), tracks[i].GetParam().GetCov(13), tracks[i].GetParam().GetCov(14) });
285  oTrack.setTime0(time0);
286  oTrack.setDeltaTBwd(tBwd);
287  oTrack.setDeltaTFwd(tFwd);
288  // RS: TODO: once the A/C merging will be implemented, both A and C side must be flagged for meged track
289  if (tracks[i].CSide()) {
290  oTrack.setHasCSideClusters();
291  } else {
292  oTrack.setHasASideClusters();
293  }
294 
295  oTrack.setChi2(tracks[i].GetParam().GetChi2());
296  auto& outerPar = tracks[i].OuterParam();
297  oTrack.setOuterParam(o2::track::TrackParCov(
298  outerPar.fX, outerPar.fAlpha,
299  { outerPar.fP[0], outerPar.fP[1], outerPar.fP[2], outerPar.fP[3], outerPar.fP[4] },
300  { outerPar.fC[0], outerPar.fC[1], outerPar.fC[2], outerPar.fC[3], outerPar.fC[4], outerPar.fC[5],
301  outerPar.fC[6], outerPar.fC[7], outerPar.fC[8], outerPar.fC[9], outerPar.fC[10], outerPar.fC[11],
302  outerPar.fC[12], outerPar.fC[13], outerPar.fC[14] }));
303  oTrack.resetClusterReferences(tracks[i].NClusters());
304  std::vector<std::pair<MCCompLabel, unsigned int>> labels;
305  for (int j = 0; j < tracks[i].NClusters(); j++) {
306  int clusterId = trackClusters[tracks[i].FirstClusterRef() + j].fNum;
307  Sector sector = clusterId >> 24;
308  int globalRow = (clusterId >> 16) & 0xFF;
309  clusterId &= 0xFFFF;
310  const ClusterNative& cl = clusters.clusters[sector][globalRow][clusterId];
311  int regionNumber = 0;
312  while (globalRow > mapper.getGlobalRowOffsetRegion(regionNumber) + mapper.getNumberOfRowsRegion(regionNumber))
313  regionNumber++;
314  CRU cru(sector, regionNumber);
315  oTrack.addCluster(Cluster(cru, globalRow - mapper.getGlobalRowOffsetRegion(regionNumber), cl.qTot, cl.qMax,
316  cl.getPad(), cl.getSigmaPad(), cl.getTime(), cl.getSigmaTime()));
317  oTrack.setClusterReference(j, sector, globalRow, clusterId);
318  if (outputTracksMCTruth) {
319  for (const auto& element : clusters.clustersMCTruth[sector][globalRow]->getLabels(clusterId)) {
320  bool found = false;
321  for (int l = 0; l < labels.size(); l++) {
322  if (labels[l].first == element) {
323  labels[l].second++;
324  found = true;
325  break;
326  }
327  }
328  if (!found)
329  labels.emplace_back(element, 1);
330  }
331  }
332  }
333  if (outputTracksMCTruth) {
334  int bestLabelNum = 0, bestLabelCount = 0;
335  for (int j = 0; j < labels.size(); j++) {
336  if (labels[j].second > bestLabelCount) {
337  bestLabelNum = j;
338  bestLabelCount = labels[j].second;
339  }
340  }
341  MCCompLabel& bestLabel = labels[bestLabelNum].first;
342  if (bestLabelCount < (1.f - sTrackMCMaxFake) * tracks[i].NClusters())
343  bestLabel.set(-bestLabel.getTrackID(), bestLabel.getEventID(), bestLabel.getSourceID());
344  outputTracksMCTruth->addElement(iTmp, bestLabel);
345  }
346  int lastSector = trackClusters[tracks[i].FirstClusterRef() + tracks[i].NClusters() - 1].fNum >> 24;
347  }
348  }
349  mTrackingCAO2Interface->Cleanup();
350  return (retVal);
351 }
