52 void Segmentation::fillRtree()
53 {
54  int paduid{ 0 };
55 
56  for (auto padGroupIndex = 0; padGroupIndex < mPadGroups.size(); ++padGroupIndex) {
57  mPadGroupIndex2PadUidIndex.push_back(paduid);
58  auto& pg = mPadGroups[padGroupIndex];
59  auto& pgt = mPadGroupTypes[pg.mPadGroupTypeId];
60  double dx{ mPadSizes[pg.mPadSizeId].first };
61  double dy{ mPadSizes[pg.mPadSizeId].second };
62  for (int ix = 0; ix < pgt.getNofPadsX(); ++ix) {
63  for (int iy = 0; iy < pgt.getNofPadsY(); ++iy) {
64  if (pgt.id(ix, iy) >= 0) {
65 
66  double xmin = ix * dx + pg.mX;
67  double xmax = (ix + 1) * dx + pg.mX;
68  double ymin = iy * dy + pg.mY;
69  double ymax = (iy + 1) * dy + pg.mY;
70 
71  mRtree.insert(std::make_pair(
72  Segmentation::Box{ Segmentation::Point(xmin, ymin), Segmentation::Point(xmax, ymax) }, paduid));
73 
74  mPadUid2PadGroupIndex.push_back(padGroupIndex);
75  mPadUid2PadGroupTypeFastIndex.push_back(pgt.fastIndex(ix, iy));
76  ++paduid;
77  }
78  }
79  }
80  }
81 }

121 {
122  std::vector<int> pi;
123

149 std::vector<int> Segmentation::getNeighbouringPadUids(int paduid) const
150 {
151  double x = padPositionX(paduid);
152  double y = padPositionY(paduid);
153  double dx = padSizeX(paduid) / 2.0;
154  double dy = padSizeY(paduid) / 2.0;
155 
156  const double offset{ 0.1 }; // 1 mm
157 
158  auto pads = getPadUids(x - dx - offset, y - dy - offset, x + dx + offset, y + dy + offset);
159  pads.erase(std::remove(begin(pads), end(pads), paduid), end(pads));
160  return pads;
161 }

170 int Segmentation::findPadByPosition(double x, double y) const
171 {
172  const double epsilon{ 1E-4 };
173  auto pads = getPadUids(x - epsilon, y - epsilon, x + epsilon, y + epsilon);
174 
175  double dmin{ std::numeric_limits<double>::max() };
176  int paduid{ InvalidPadUid };
177 
178  for (auto i = 0; i < pads.size(); ++i) {
179  double d{ squaredDistance(pads[i], x, y) };
180  if (d < dmin) {
181  paduid = pads[i];
182  dmin = d;
183  }
184  }
185 
186  return paduid;
187 }
