47 int LegacyUtility::convertFromLegacyDeId(int detElemId)
48 {
51  if (assertLegacyDeId(detElemId)) {
52  int ich = (detElemId / 100 - 11);

169 std::vector<ColumnData> LegacyUtility::digitsToPattern(std::vector<uint32_t> digits)
170 {
172  int detElemId, boardId, channel, cathode, icolumn, iline;
173  int deId;
174  std::vector<ColumnData> columns;

177  boardToPattern(boardId, detElemId, cathode, deId, icolumn, iline);
178  ColumnData* currentColumn = nullptr;
179  for (auto& column : columns) {

188  }
