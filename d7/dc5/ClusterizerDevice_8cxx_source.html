42 
43  LOG(INFO) << "Received data of size: " << msg->GetSize();
44 
45  // FIXME: This will change when we will have the final format
46  // The HLT adds an header. After 100 bytes, we have the number of digits
47  if (msg->GetSize() < 100) {
48  LOG(INFO) << "Empty message: skip";
49  return true;
50  }
51  uint8_t* data = reinterpret_cast<uint8_t*>(msg->GetData());
52  uint32_t* digitsData = reinterpret_cast<uint32_t*>(data + 100);
53  uint32_t nDigits = digitsData[0];
54  uint32_t offset(1);
55  // Loop on digits
56  std::vector<uint32_t> digits;
57  for (int idig = 0; idig < nDigits; idig++) {
58  uint32_t uniqueID = digitsData[offset++];
59  digits.push_back(uniqueID);
60  ++offset; // skip idx and adc
61  }
62  std::vector<ColumnData> patterns = LegacyUtility::digitsToPattern(digits);
63  // end of the block that will require a change
64 
65  mClusterizer.process(patterns);
66 
67  if (mClusterizer.getNClusters() > 0) {
68  // Store clusters
69  FairMQMessagePtr msgOut(NewMessage());
70  // Serialize<BoostSerializer<Track>>(*msgOut, mTracks);

72

75  LOG(ERROR) << "problem sending message";

77  }
78  }
