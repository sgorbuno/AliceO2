66  if (std::string(type).compare("TH1F") == 0) {

68  return impl::createProducer<ProducerType>(type, std::forward<Args>(args)...);

70  using ProducerType = typename impl::conditional<TH2Producer, 3, Args...>::type;
71  return impl::createProducer<ProducerType>(type, std::forward<Args>(args)...);
72  } else if (std::string(type).compare("TH3F") == 0) {
73  using ProducerType = typename impl::conditional<TH3Producer, 3, Args...>::type;
74  return impl::createProducer<ProducerType>(type, std::forward<Args>(args)...);
75  } else if (std::string(type).compare("THnF") == 0) {
76  using ProducerType = typename impl::conditional<THnProducer, 3, Args...>::type;
77  return impl::createProducer<ProducerType>(type, std::forward<Args>(args)...);
78  } else if (std::string(type).compare("TTree") == 0) {

82  LOG(ERROR) << "Unknown type of producer: " << type
