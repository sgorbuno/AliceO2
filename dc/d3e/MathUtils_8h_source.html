67 inline float MathUtils::computeCurvature(float x1, float y1, float x2, float y2, float x3, float y3)
68 {
69  const float d = (x2 - x1) * (y3 - y2) - (x3 - x2) * (y2 - y1);
70  const float a =
71  0.5f * ((y3 - y2) * (y2 * y2 - y1 * y1 + x2 * x2 - x1 * x1) - (y2 - y1) * (y3 * y3 - y2 * y2 + x3 * x3 - x2 * x2));
72  const float b =
73  0.5f * ((x2 - x1) * (y3 * y3 - y2 * y2 + x3 * x3 - x2 * x2) - (x3 - x2) * (y2 * y2 - y1 * y1 + x2 * x2 - x1 * x1));
74 
75  return -1.f * d / std::sqrt((d * x1 - a) * (d * x1 - a) + (d * y1 - b) * (d * y1 - b));
76 }

78 inline float MathUtils::computeCurvatureCentreX(float x1, float y1, float x2, float y2, float x3, float y3)
79 {
80  const float k1 = (y2 - y1) / (x2 - x1), k2 = (y3 - y2) / (x3 - x2);
81  return 0.5f * (k1 * k2 * (y1 - y3) + k2 * (x1 + x2) - k1 * (x2 + x3)) / (k2 - k1);
82 }
