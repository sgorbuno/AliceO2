80 {

119 void Clusterizer::preClusterizeNBP(PatternStruct& de)
120 {
122  PreCluster* pc = nullptr;
123  double limit = 0;
124  for (int icolumn = 0; icolumn < 7; ++icolumn) {
125  if (de.columns[icolumn].getNonBendPattern() == 0) {
126  continue;
127  }
128  int nStripsNBP = mMapping.getNStripsNBP(icolumn, de.deId);
129  for (int istrip = 0; istrip < nStripsNBP; ++istrip) {
130  if (de.columns[icolumn].isNBPStripFired(istrip)) {
131  if (pc) {
132  if (istrip == 0) {
133  // We're changing column
134  // In principle we could simply add the pitch, but for the cut RPCs
135  // the y dimension changes as well in column 0
136  pc->area[icolumn] = mMapping.stripByLocation(istrip, 1, 0, icolumn, de.deId);
137  limit = pc->area[icolumn].getXmax();
138  } else {
139  limit += mMapping.getStripSize(istrip, 1, icolumn, de.deId);
140  }
141  } else {
142  pc = nextPreCluster(7);
143  pc->paired = 0;
144  pc->firstColumn = icolumn;
145  pc->area[icolumn] = mMapping.stripByLocation(istrip, 1, 0, icolumn, de.deId);
146  limit = pc->area[icolumn].getXmax();
147  LOG(DEBUG) << "New precluster NBP: DE " << de.deId;
148  }
149  pc->lastColumn = icolumn;
150  pc->area[icolumn].setXmax(limit);
151  LOG(DEBUG) << " adding col " << icolumn << " strip " << istrip << " (" << pc->area[icolumn].getXmin() << ", "
152  << pc->area[icolumn].getXmax() << ") (" << pc->area[icolumn].getYmin() << ", "
153  << pc->area[icolumn].getYmax() << ")";
154  } else {
155  pc = nullptr;
156  }
157  }
158  de.columns[icolumn].setNonBendPattern(0); // Reset pattern
159  }

163 void Clusterizer::preClusterizeBP(PatternStruct& de)
164 {
166  double limit = 0;
167  for (int icolumn = mMapping.getFirstColumn(de.deId); icolumn < 7; ++icolumn) {
168  if ((de.firedColumns & (1 << icolumn)) == 0) {
169  continue;
170  }
171  PreCluster* pc = nullptr;
172  int firstLine = mMapping.getFirstBoardBP(icolumn, de.deId);
173  int lastLine = mMapping.getLastBoardBP(icolumn, de.deId);
174  for (int iline = firstLine; iline <= lastLine; ++iline) {
175  if (de.columns[icolumn].getBendPattern(iline) == 0) {
176  continue;
177  }
178  for (int istrip = 0; istrip < 16; ++istrip) {
179  if (de.columns[icolumn].isBPStripFired(istrip, iline)) {
180  if (pc) {
181  limit += mMapping.getStripSize(istrip, 0, icolumn, de.deId);
182  } else {
183  pc = nextPreCluster(icolumn);
184  pc->paired = 0;
185  pc->firstColumn = icolumn;
186  pc->lastColumn = icolumn;
187  pc->area[icolumn] = mMapping.stripByLocation(istrip, 0, iline, icolumn, de.deId);
188  limit = pc->area[icolumn].getYmax();
189  LOG(DEBUG) << "New precluster BP: DE " << de.deId << " icolumn " << icolumn;
190  }
191  pc->area[icolumn].setYmax(limit);
192  LOG(DEBUG) << " adding line " << iline << " strip " << istrip << " (" << pc->area[icolumn].getXmin()
193  << ", " << pc->area[icolumn].getXmax() << ") (" << pc->area[icolumn].getYmin() << ", "
194  << pc->area[icolumn].getYmax() << ")";
195  } else {
196  pc = nullptr;
197  }
198  }
199  de.columns[icolumn].setBendPattern(0, iline); // Reset pattern
200  } // loop on lines
201  } // loop on columns
202 }

205 void Clusterizer::makeClusters(const int& deIndex)
206 {
208  LOG(DEBUG) << "Clusterizing " << deIndex;
209 
210  // loop over pre-clusters in the non-bending plane
211  for (int inb = 0; inb < mNPreClusters[7]; ++inb) {
212  PreCluster& pcNB = mPreClusters[7][inb];
213  int icolumn = pcNB.firstColumn;
214  if (icolumn == pcNB.lastColumn) {
215  // This is the most simple and common case: the NBP pre-cluster is on
216  // one single column. So it can be easily matched with the BP
217  // since the corresponding contours are both rectangles
218  for (int ib = 0; ib < mNPreClusters[icolumn]; ++ib) {
219  PreCluster& pcB = mPreClusters[icolumn][ib];
220  makeCluster(pcB, pcNB, deIndex);
221  }
222  } else {
223  // The NBP pre-cluster spans different columns.
224  // The BP contour is therefore a serie of neighbour rectangles
225  std::vector<std::vector<PreCluster*>> pcBneighbours;
226  LOG(DEBUG) << "Spanning non-bend: " << icolumn << " -> " << pcNB.lastColumn;
227  buildListOfNeighbours(icolumn, pcNB.lastColumn, pcBneighbours);
228  for (auto& pcBlist : pcBneighbours) {
229  makeCluster(pcBlist, deIndex, &pcNB);
230  }
231  }
232 
233  if (pcNB.paired > 0) {
234  continue;
235  }
236  // If it is not paired, it means that we have
237  // a monocathodic cluster in the NBP
238  makeCluster(pcNB, pcNB, deIndex);
239  } // loop over pre-clusters in the NBP
240 
242  std::vector<std::vector<PreCluster*>> pcBneighbours;
243  buildListOfNeighbours(0, 6, pcBneighbours, true);
244  for (auto& pcBlist : pcBneighbours) {
245  makeCluster(pcBlist, deIndex);
246  }
247 }

262 void Clusterizer::makeCluster(PreCluster& clBend, PreCluster& clNonBend, const int& deIndex)
263 {
265  Cluster2D& cl = nextCluster();
266  int icolumn = clBend.firstColumn;
267  cl.deId = (uint8_t)deIndex;
268  cl.xCoor = 0.5 * (clNonBend.area[icolumn].getXmax() + clNonBend.area[icolumn].getXmin());
269  cl.yCoor = 0.5 * (clBend.area[icolumn].getYmax() + clBend.area[icolumn].getYmin());
270  double deltaX = clNonBend.area[icolumn].getXmax() - clNonBend.area[icolumn].getXmin();
271  double deltaY = clBend.area[icolumn].getYmax() - clBend.area[icolumn].getYmin();
272  cl.sigmaX2 = deltaX * deltaX / 12;
273  cl.sigmaY2 = deltaY * deltaY / 12;
274  clBend.paired = 1;
275  clNonBend.paired = 1;
276 
277  LOG(DEBUG) << "pos: (" << cl.xCoor << ", " << cl.yCoor << ") err2: (" << cl.sigmaX2 << ", " << cl.sigmaY2 << ")";
278 }

281 void Clusterizer::makeCluster(std::vector<PreCluster*>& pcBlist, const int& deIndex, PreCluster* clNonBend)
282 {
284  // This is the general case:
285  // perform the full calculation assuming a uniform charge distribution
286 
287  if (pcBlist.size() == 1) {
288  // We fall back to the simple case:
289  PreCluster* pcBP = pcBlist[0];
290  PreCluster* pcNBP = clNonBend ? clNonBend : pcBP;
291  makeCluster(*pcBP, *pcNBP, deIndex);
292  return;
293  }
294 
295  Cluster2D& cl = nextCluster();
296  cl.deId = (uint8_t)deIndex;
297 
298  double x2[2][2] = { { 0., 0. }, { 0., 0. } };
299  double x3[2][2] = { { 0., 0. }, { 0., 0. } };
300  double dim[2][2];
301  double delta[2];
302  double sumArea = 0.;
303 
304  for (auto* pcBP : pcBlist) {
305  PreCluster* pcNBP = clNonBend ? clNonBend : pcBP;
306  int icolumn = pcBP->firstColumn;
307  dim[0][0] = pcNBP->area[icolumn].getXmin();
308  dim[0][1] = pcNBP->area[icolumn].getXmax();
309  dim[1][0] = pcBP->area[icolumn].getYmin();
310  dim[1][1] = pcBP->area[icolumn].getYmax();
311  for (int iplane = 0; iplane < 2; ++iplane) {
312  delta[iplane] = dim[iplane][1] - dim[iplane][0];
313  }
314  // area = dx * dy
315  sumArea += delta[0] * delta[1];
316  LOG(DEBUG) << "Area += " << delta[0] * delta[1] << " => " << sumArea;
317  for (int iplane = 0; iplane < 2; ++iplane) {
318  for (int ip = 0; ip < 2; ++ip) {
319  // second momentum = x_i * x_i * dy
320  double currX2 = dim[iplane][ip] * dim[iplane][ip] * delta[1 - iplane];
321  x2[iplane][ip] += currX2;
322  // third momentum = x_i * x_i * x_i * dy
323  x3[iplane][ip] += currX2 * dim[iplane][ip];
324  LOG(DEBUG) << "x[" << iplane << "][" << ip << "] => val " << dim[iplane][ip] << " delta " << delta[1 - iplane]
325  << " => x2 " << x2[iplane][ip] << " x3 " << x3[iplane][ip];
326  }
327  }
328  pcBP->paired = 1;
329  pcNBP->paired = 1;
330  } // loop on column
331 
332  double coor[2], sigma2[2];
333  for (int iplane = 0; iplane < 2; ++iplane) {
334  coor[iplane] = (x2[iplane][1] - x2[iplane][0]) / sumArea / 2.;
335  sigma2[iplane] = (x3[iplane][1] - x3[iplane][0]) / sumArea / 3. - coor[iplane] * coor[iplane];
336  }
337 
338  cl.xCoor = (float)coor[0];
339  cl.yCoor = (float)coor[1];
340  cl.sigmaX2 = (float)sigma2[0];
341  cl.sigmaY2 = (float)sigma2[1];
342 
343  LOG(DEBUG) << "pos: (" << cl.xCoor << ", " << cl.yCoor << ") err2: (" << cl.sigmaX2 << ", " << cl.sigmaY2 << ")";
344 }

347 bool Clusterizer::buildListOfNeighbours(int icolumn, int lastColumn, std::vector<std::vector<PreCluster*>>& neighbours,
348  bool skipPaired, int currentList)
349 {
351  LOG(DEBUG) << "Building list of neighbours in (" << icolumn << ", " << lastColumn << ")";
352  for (int jcolumn = icolumn; jcolumn <= lastColumn; ++jcolumn) {
353  for (int ib = 0; ib < mNPreClusters[jcolumn]; ++ib) {
354  PreCluster* pcB = &mPreClusters[jcolumn][ib];
355  if (skipPaired && pcB->paired > 0) {
356  LOG(DEBUG) << "Column " << jcolumn << " ib " << ib << " is already paired => skipPaired";
357  continue;
358  }
359  if (currentList >= neighbours.size()) {
360  // We are starting a new series of neighbour
361  // Let's make sure the pre-cluster is not already part of another list
362  if (pcB->paired == 2) {
363  LOG(DEBUG) << "Column " << jcolumn << " ib " << ib << " is already in a list";
364  continue;
365  }
366  LOG(DEBUG) << "New list " << currentList;
367  neighbours.emplace_back(std::vector<PreCluster*>());
368  }
369  std::vector<PreCluster*>& neighList = neighbours[currentList];
370  if (!neighList.empty()) {
371  auto* neigh = neighList.back();
372  if (neigh->area[jcolumn - 1].getYmin() > pcB->area[jcolumn].getYmax())
373  continue;
374  if (neigh->area[jcolumn - 1].getYmax() < pcB->area[jcolumn].getYmin())
375  continue;
376  }
377  pcB->paired = 2;
378  LOG(DEBUG) << " adding column " << jcolumn << " ib " << ib << " to " << currentList;
379  neighList.push_back(pcB);
380  buildListOfNeighbours(jcolumn + 1, lastColumn, neighbours, skipPaired, currentList);
381  ++currentList;
382  }
383  }
384 
385  return (neighbours.size() > 0);
386 }
