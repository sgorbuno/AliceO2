32 int LookUp::groupFinder(int nRow, int nCol)
33 {
34  int row_index = nRow / TopologyDictionary::RowClassSpan;
35  if (nRow % TopologyDictionary::RowClassSpan == 0) {
36  row_index--;
37  }
38  int col_index = nCol / TopologyDictionary::ColClassSpan;
39  if (nCol % TopologyDictionary::RowClassSpan == 0) {

41  }

50 {
